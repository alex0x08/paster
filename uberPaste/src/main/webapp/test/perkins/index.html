// expects: @weight, @size, @lineheight, @fontfamily
// defaults: normal, 16px, 20px, @sans
// note that the variable @sans can be changed in perkins.config.less

// DEMO:
.p-fontmixin {
  .font(bold, 28px, 32px, "Georgia", Times New Roman, serif);
  text-align: center;
